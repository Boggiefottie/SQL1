-- INNER JOIN
SELECT  C.FIRST_NAME,O.ORDER_ID,O.AMOUNT 
FROM CUSTOMER C INNER JOIN ORDERS O ON C.CUSTOMER_ID=O.CUSTOMER_ID;
 
--LEFT OUTER JOIN

SELECT  C.FIRST_NAME,O.ORDER_ID,O.AMOUNT 
FROM CUSTOMER C LEFT OUTER JOIN ORDERS O ON C.CUSTOMER_ID=O.CUSTOMER_ID;
 
-- RIGHT OUTER JOIN
SELECT  C.FIRST_NAME,O.ORDER_ID,O.AMOUNT 
FROM CUSTOMER C RIGHT OUTER JOIN ORDERS O ON C.CUSTOMER_ID=O.CUSTOMER_ID;
 
 
--FULL OUTER JOIN
SELECT  C.FIRST_NAME,O.ORDER_ID,O.AMOUNT 
FROM CUSTOMER C FULL OUTER JOIN ORDERS O ON C.CUSTOMER_ID=O.CUSTOMER_ID;
