--List employees belonging to department 30, 40, or 10 
SELECT EMP_ID,EMP_NAME
FROM EMP  
WHERE DEPT_ID=30 OR DEPT_ID=40 OR DEPT_ID=10; 

--3. List the employee details whose salary is between 10000 to 30000.

SELECT EMP_NAME,EMP_SAL
FROM EMP
WHERE EMP_SAL>=10000 AND EMP_SAL<=30000;

-- List total no of employee
SELECT COUNT(EMP_ID) AS TOTAL_EMP
FROM EMP;

--List average sal of each deptID. 
SELECT DEPT_ID,AVG(EMP_SAL)AS AVERAGE_SALARY
FROM EMP
GROUP BY DEPT_ID;

--List employee details in ascending order of salary
SELECT EMP_ID,EMP_NAME,EMP_SAL
FROM EMP
ORDER BY EMP_SAL ;
